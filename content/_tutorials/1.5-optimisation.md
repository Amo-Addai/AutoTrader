```python
from autotrader.auto_optimise import Optimise

# Optimise strategy parameters
opt                 = Optimise()
opt.config_file     = 'stock_ema_crossover'
opt.opt_params      = ['slow_ema', 'fast_ema']
opt.bounds          = [(5, 30), (10, 50)]
opt.Ns              = 8

opt.optimise()
```
